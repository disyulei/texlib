% ============================================ 
%              Rectangle Box
% ============================================ 
% Example:  \myrect{\pgfxy(1,1)}{\pgfxy(2,3)}
\newcommand{\myrect}[2]
{
  {
     \color{UtdaOrange}        % set boader color
     \pgfsetlinewidth{2pt}     % set boader width
     \pgfrect[stroke]{#1}{#2}
     \color{structure}         % set boader color
   }
}


% ============================================ 
%                 texting
% ============================================ 
% Draw red (orange) color text
% Example:  \myredtext{\pgfxy(1,1)}{text to print}
\newcommand{\myredtext}[2]
{
     \pgfputat{#1}
     {
         \pgfbox[center,base]
         {\color{UtdaOrange}#2}
     }
}
% Draw blue color text
\newcommand{\mybluetext}[2]
{
     \pgfputat{#1}
     {
         \pgfbox[center,base]{\structure{#2}}
     }
}
\newcommand{\mytext}[2]{\myredtext{#1}{#2}}


% ============================================ 
%             Connecting Arrow
% ============================================ 
\newcommand{\smallconnect}[4]{%
   \pgfsetlinewidth{1.5pt}%
   %\color{structure}%
   \color{orange}
   %\color{blue!70}
   \pgfsetendarrow{\pgfarrowto}
   \pgfnodeconncurve{#1}{#2}{#3}{#4}{1cm}{1cm}%
   \color{structure}
}
